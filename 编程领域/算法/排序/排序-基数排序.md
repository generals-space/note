基数排序的工作流程可以查看参考文章2和参考文章4中的配图.

从个位开始, 遍历待排序数组依次填充. 排到最高位时, 就可以保证所有成员有序了.

```js
var arr = [5, 38, 15, 48, 44, 3, 36, 26, 50, 27, 2, 46, 4, 19, 47];

console.log(arr);
arr = radixSort(arr, 2);
console.log(arr);

/**
 * @function 
 * @param {*} arr 
 * @param {*} maxDigit 待排序数组中最大值的位数. 如最大值为10000, 那么此值应为5(5位数).
 */
function radixSort(arr, maxDigit) {
    let counter = [];

    // 从个位开始, 遍历数组中的元素并排序
    for (let i = 0; i < maxDigit; i++) {
        let mod = Math.pow(10, i+1); // 取余
        let dev = Math.pow(10, i); // 除数
        for(let j = 0; j < arr.length; j++) {
            // val为当前数组成员的个位/十位/百位..的数值
            // arr[j] % mod / dev 可以得到一个小数
            let val = parseInt((arr[j] % mod) / dev);
            if(counter[val] == null) {
                counter[val] = [];
            }
            counter[val].push(arr[j]);
        }

        // 将存入counter各数位的子数组取出, 准备对下一位数据的排序.
        let pos = 0;
        for(let j = 0; j < counter.length; j++) {
            if(!counter[j]) continue;

            let val = null;
            while ((val = counter[j].shift()) != null) {
                arr[pos] = val;
                pos++;
            }
        }
    }
    return arr;
}

```

```py
import math

arr = [5, 38, 15, 48, 44, 3, 36, 26, 50, 27, 2, 46, 4, 19, 47]

def radix_sort(arr, maxDigit):

    for i in range(0, maxDigit):
        counter = [[] for i in range(0, 10)]
        mod = math.pow(10, i+1)
        dev = math.pow(10, i)
        for j in range(0, len(arr)):
            val = math.floor(arr[j] % mod / dev)
            counter[val].append(arr[j])

        pos = 0
        for i in range(0, len(counter)):
            if len(counter[i]) == 0: continue

            while True:
                if len(counter[i]) == 0: break
                val = counter[i][0]
                arr[pos] = val
                pos += 1
                counter[i] = counter[i][1:]

    return arr

print(arr)
arr = radix_sort(arr, 2)
print(arr)
```
